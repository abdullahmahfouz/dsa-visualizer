What is a Hash Table with Linear Probing?

A Hash Table is a data structure that stores key-value pairs. It uses a hash function to compute an index into an array, allowing for fast lookups.

Think of a library where books are sorted by genre (the hash) for faster retrieval.

Collisions and Linear Probing

A collision occurs when two different keys hash to the same index. Linear Probing is a simple way to resolve collisions. If a collision occurs, it checks the next available slot in the array and places the new key there. This process continues until an empty slot is found.

import java.util.Arrays;

class HashTable {
    private static final int DEFAULT_CAPACITY = 10;
    private Entry[] table;
    private int size;

    static class Entry {
        Object key;
        Object value;
        boolean isDeleted;

        Entry(Object key, Object value) {
            this.key = key;
            this.value = value;
            this.isDeleted = false;
        }
    }

    public HashTable() {
        table = new Entry[DEFAULT_CAPACITY];
        size = 0;
    }

    private int hash(Object key) {
        return Math.abs(key.hashCode() % table.length);
    }

    public void insert(Object key, Object value) {
        int index = hash(key);
        int startIndex = index;
        while (table[index] != null && !table[index].isDeleted) {
            if (table[index].key.equals(key)) {
                table[index].value = value; // Update existing key
                return;
            }
            index = (index + 1) % table.length;
            if (index == startIndex) { // Table is full
                System.err.println("Hash table is full.");
                return;
            }
        }
        table[index] = new Entry(key, value);
        size++;
    }

    public Object search(Object key) {
        int index = hash(key);
        int startIndex = index;
        while (table[index] != null) {
            if (!table[index].isDeleted && table[index].key.equals(key)) {
                return table[index].value;
            }
            index = (index + 1) % table.length;
            if (index == startIndex) { // Cycled through table
                break;
            }
        }
        return null;
    }

    public void delete(Object key) {
        int index = hash(key);
        int startIndex = index;
        while (table[index] != null) {
            if (!table[index].isDeleted && table[index].key.equals(key)) {
                table[index].isDeleted = true;
                size--;
                return;
            }
            index = (index + 1) % table.length;
            if (index == startIndex) { // Cycled through table
                break;
            }
        }
    }
}

class HashTable {

    std::vector<std::optional<Entry>> table;
    size_t table_size;
    size_t count = 0;

    size_t hash(const K& key) {
        return std::hash<K>{}(key) % table_size;
    }

public:
    HashTable(size_t size = 10) : table_size(size) {
        table.resize(table_size);
    }

    void insert(const K& key, const V& value) {
        if (count >= table_size) {
            // In a real implementation, you would resize the table here
            std::cerr << "Table is full!" << std::endl;
            return;
        }
        size_t index = hash(key);
        while (table[index] && !table[index]->is_deleted) {
            if (table[index]->key == key) {
                table[index]->value = value; // Update
                return;
            }
            index = (index + 1) % table_size;
        }
        table[index] = Entry{key, value};
        count++;
    }

    std::optional<V> search(const K& key) {
        size_t index = hash(key);
        size_t start_index = index;
        while (table[index]) {
            if (!table[index]->is_deleted && table[index]->key == key) {
                return table[index]->value;
            }
            index = (index + 1) % table_size;
            if (index == start_index) break;
        }
        return std::nullopt;
    }

    void remove(const K& key) {
        size_t index = hash(key);
        size_t start_index = index;
        while (table[index]) {
            if (!table[index]->is_deleted && table[index]->key == key) {
                table[index]->is_deleted = true;
                count--;
                return;
            }
            index = (index + 1) % table_size;
            if (index == start_index) break;
        }
    }
};

#define TABLE_SIZE 10

typedef struct {
    char* key;
    int value;
    int is_deleted;
} Entry;

Entry* table[TABLE_SIZE];

unsigned int hash(char* key) {
    unsigned int hash = 0;
    while (*key) {
        hash = (hash << 5) + *key++;
    }
    return hash % TABLE_SIZE;
}

void init_table() {
    for (int i = 0; i < TABLE_SIZE; i++) {
        table[i] = NULL;
    }
}

void insert(char* key, int value) {
    unsigned int index = hash(key);
    int original_index = index;
    while (table[index] != NULL && !table[index]->is_deleted) {
        if (strcmp(table[index]->key, key) == 0) {
            table[index]->value = value; // Update
            return;
        }
        index = (index + 1) % TABLE_SIZE;
        if (index == original_index) {
            printf("Error: Hash table is full.\n");
            return;
        }
    }
    table[index] = (Entry*)malloc(sizeof(Entry));
    table[index]->key = strdup(key);
    table[index]->value = value;
    table[index]->is_deleted = 0;
}

Entry* search(char* key) {
    unsigned int index = hash(key);
    int original_index = index;
    while (table[index] != NULL) {
        if (!table[index]->is_deleted && strcmp(table[index]->key, key) == 0) {
            return table[index];
        }
        index = (index + 1) % TABLE_SIZE;
        if (index == original_index) break;
    }
    return NULL;
}

void delete_entry(char* key) {
    unsigned int index = hash(key);
    int original_index = index;
    while (table[index] != NULL) {
        if (!table[index]->is_deleted && strcmp(table[index]->key, key) == 0) {
            table[index]->is_deleted = 1;
