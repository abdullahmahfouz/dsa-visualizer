Stack is a LIFO structure:

  • Push O(1): Add to top
  • Pop O(1): Remove from top
  • Peek O(1): View top element

class Stack:
    def __init__(self):
        self.items = []
    
    def push(self, item):
        """Add item to top - O(1)"""
        self.items.append(item)
    
    def pop(self):
        """Remove and return top item - O(1)"""
        if not self.is_empty():
            return self.items.pop()
        raise IndexError("Stack is empty")
    
    def peek(self):
        """View top item without removing - O(1)"""
        if not self.is_empty():
            return self.items[-1]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)

            return this.items.pop();
        }

    }

        // View top item without removing - O(1)
        if (!this.isEmpty()) {
            return this.items[this.items.length - 1];
        }
        return null;
    }
    
    isEmpty() {
        return this.items.length === 0;
    }
    
    size() {
        return this.items.length;
    }
}

import java.util.ArrayList;

public class Stack<T> {
    private ArrayList<T> items;
    
    public Stack() {
        this.items = new ArrayList<>();
    }
    
    // Add item to top - O(1)
    public void push(T item) {
        items.add(item);
    }
    
    // Remove and return top item - O(1)
    public T pop() {
        if (!isEmpty()) {
            return items.remove(items.size() - 1);
        }
        throw new RuntimeException("Stack is empty");
    }
    
    // View top item without removing - O(1)
    public T peek() {
        if (!isEmpty()) {
            return items.get(items.size() - 1);
        }
        return null;
    }
    
    public boolean isEmpty() {
        return items.isEmpty();
    }
    
    public int size() {
        return items.size();
    }
}

public class Stack<T> {
    private List<T> items = new List<T>();
    
    // Add item to top - O(1)
    public void Push(T item) {
        items.Add(item);
    }
    
    // Remove and return top item - O(1)
    public T Pop() {
        if (!IsEmpty()) {
            T item = items[items.Count - 1];
            items.RemoveAt(items.Count - 1);
            return item;
        }
        throw new InvalidOperationException("Stack is empty");
    }
    
    // View top item without removing - O(1)
    public T Peek() {
        if (!IsEmpty()) {
            return items[items.Count - 1];
        }
        return default(T);
    }
    
    public bool IsEmpty() => items.Count == 0;
    
    public int Size() => items.Count;
}

template <typename T>
class Stack {
private:
    std::vector<T> items;

public:
    // Add item to top - O(1)
    void push(T item) {
        items.push_back(item);
    }
    
    // Remove and return top item - O(1)
    T pop() {
        if (!isEmpty()) {
            T item = items.back();
            items.pop_back();
            return item;
        }
        throw std::runtime_error("Stack is empty");
    }
    
    // View top item without removing - O(1)
    T peek() {
        if (!isEmpty()) {
            return items.back();
        }
        throw std::runtime_error("Stack is empty");
    }
    
    bool isEmpty() {
        return items.empty();
    }
    
    int size() {
        return items.size();
    }
};

#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

#define MAX_SIZE 100

typedef struct {
    int items[MAX_SIZE];
    int top;
} Stack;

void init(Stack* s) {
    s->top = -1;
}

// Add item to top - O(1)
void push(Stack* s, int item) {
    if (s->top < MAX_SIZE - 1) {
        s->items[++s->top] = item;
    }
}

// Remove and return top item - O(1)
int pop(Stack* s) {
    if (!isEmpty(s)) {
        return s->items[s->top--];
    }
    return -1; // Error
}

// View top item without removing - O(1)
int peek(Stack* s) {
    if (!isEmpty(s)) {
        return s->items[s->top];
    }
    return -1; // Error
}

bool isEmpty(Stack* s) {
    return s->top == -1;
}

int size(Stack* s) {
    return s->top + 1;
}

// Usage
int main() {
    Stack stack;
    init(&stack);
    push(&stack, 10);
    push(&stack, 20);
    printf("%d", pop(&stack)); // 20
    return 0;
}
