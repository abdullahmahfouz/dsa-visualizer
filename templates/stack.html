What is a Stack?

A Stack is a linear data structure that follows the LIFO (Last In, First Out) principle.

Think of a stack of plates - you can only add or remove from the top!
Push Add to top O(1)
Pop Remove from top O(1)
Peek View top element O(1)
Common Uses:
  • Browser back/forward buttons
  • Undo/Redo functionality
  • Function call stack
  • Expression evaluation

class Stack:
    def __init__(self, capacity=100):
        self.capacity = capacity
        self.items = [None] * capacity  # Fixed-size array
        self.top = -1  # Index of top element
    
    def push(self, item):
        """Add item to top - O(1)"""
        if self.top >= self.capacity - 1:
            raise OverflowError("Stack is full")
        self.top += 1
        self.items[self.top] = item
    
    def pop(self):
        """Remove and return top item - O(1)"""
        if self.is_empty():
            raise IndexError("Stack is empty")
        item = self.items[self.top]
        self.items[self.top] = None
        self.top -= 1
        return item
    
    def peek(self):
        """View top item without removing - O(1)"""
        if self.is_empty():
            return None
        return self.items[self.top]
    
    def is_empty(self):
        return self.top == -1
    
    def size(self):
        return self.top + 1

class Stack {
    constructor(capacity = 100) {
        this.capacity = capacity;
        this.items = new Array(capacity);  // Fixed-size array
        this.top = -1;  // Index of top element
    }
    
    push(item) {
        // Add item to top - O(1)
        if (this.top >= this.capacity - 1) {
            throw new Error("Stack is full");
        }
        this.top++;
        this.items[this.top] = item;
    }
    
    pop() {
        // Remove and return top item - O(1)
        if (this.isEmpty()) {
            throw new Error("Stack is empty");
        }
        const item = this.items[this.top];
        this.items[this.top] = undefined;
        this.top--;
        return item;
    }
    
    peek() {
        // View top item without removing - O(1)
        if (this.isEmpty()) {
            return null;
        }
        return this.items[this.top];
    }
    
    isEmpty() {
        return this.top === -1;
    }
    
    size() {
        return this.top + 1;
    }
}

#include <stdexcept>

class Stack {
private:
    int* items;    // Fixed-size array
    int top;       // Index of top element
    int capacity;

public:
    Stack(int cap = 100) : capacity(cap), top(-1) {
        items = new int[capacity];
    }
    
    ~Stack() { delete[] items; }
    
    // Add item to top - O(1)
    void push(int item) {
        if (top >= capacity - 1)
            throw std::overflow_error("Stack is full");
        items[++top] = item;
    }
    
    // Remove and return top item - O(1)
    int pop() {
        if (isEmpty())
            throw std::underflow_error("Stack is empty");
        return items[top--];
    }
    
    // View top item without removing - O(1)
    int peek() {
        if (isEmpty())
            throw std::underflow_error("Stack is empty");
        return items[top];
    }
    
    bool isEmpty() { return top == -1; }
    
    int size() { return top + 1; }
};

// Usage
int main() {
    Stack stack(100);
    stack.push(10);
    stack.push(20);
    std::cout << stack.pop(); // 20
    return 0;
}
#include <stdio.h>
#include <stdbool.h>

#define MAX_SIZE 100

typedef struct {
    int items[MAX_SIZE];  // Fixed-size array
    int top;              // Index of top element
} Stack;

void init(Stack* s) {
    s->top = -1;
}

// Add item to top - O(1)
bool push(Stack* s, int item) {
    if (s->top >= MAX_SIZE - 1)
        return false;  // Stack overflow
    s->items[++s->top] = item;
    return true;
}

// Remove and return top item - O(1)
int pop(Stack* s) {
    if (isEmpty(s))
        return -1;  // Stack underflow
    return s->items[s->top--];
}

// View top item without removing - O(1)
int peek(Stack* s) {
    if (isEmpty(s))
        return -1;
    return s->items[s->top];
}

bool isEmpty(Stack* s) {
    return s->top == -1;
}

int size(Stack* s) {
    return s->top + 1;
}

// Usage
int main() {
    Stack stack;
    init(&stack);
    push(&stack, 10);
    push(&stack, 20);
    printf("%d", pop(&stack)); // 20
    return 0;
}
