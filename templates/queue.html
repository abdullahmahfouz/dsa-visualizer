What is a Queue?

A Queue is a linear data structure that follows the FIFO (First In, First Out) principle. Elements are added at the back and removed from the front. This structure is ideal for scenarios where order matters and the first element added should be the first one removed.

Think of a line at a store – the first person in line gets served first!
Enqueue Add to the back O(1)
Dequeue Remove from the front O(1)
Peek View the front element O(1)
Clear Remove all elements O(1)
Common Uses:
  • Printer job scheduling
  • Order processing systems
  • Breadth-first search in graphs
  • Task scheduling in operating systems

class Queue:
    """Queue using Python's dynamic list - simple and Pythonic!"""
    
    def __init__(self):
        self.items = []  # Dynamic list, no fixed size
    
    def enqueue(self, item):
        """Add item to back - O(1) amortized"""
        self.items.append(item)
    
    def dequeue(self):
        """Remove and return front item - O(n)"""
        if self.is_empty():
            raise IndexError("Cannot dequeue from empty queue")
        return self.items.pop(0)  # Remove from front
    
    def peek(self):
        """View front item without removing - O(1)"""
        if self.is_empty():
            return None
        return self.items[0]
    
    def is_empty(self):
        """Check if queue is empty - O(1)"""
        return len(self.items) == 0
    
    def size(self):
        """Get number of items - O(1)"""
        return len(self.items)

class Queue {
    constructor(capacity = 100) {
        this.capacity = capacity;
        this.items = new Array(capacity);  // Fixed-size array
        this.front = 0;   // Index of front element
        this.rear = -1;   // Index of rear element
        this.count = 0;   // Number of elements
    }
    
    enqueue(item) {
        // Add item to back - O(1)
        if (this.count >= this.capacity) {
            throw new Error("Queue is full");
        }
        this.rear = (this.rear + 1) % this.capacity;
        this.items[this.rear] = item;
        this.count++;
    }
    
    dequeue() {
        // Remove and return front item - O(1)
        if (this.isEmpty()) {
            throw new Error("Queue is empty");
        }
        const item = this.items[this.front];
        this.items[this.front] = undefined;
        this.front = (this.front + 1) % this.capacity;
        this.count--;
        return item;
    }
    
    peek() {
        // View front item without removing - O(1)
        if (this.isEmpty()) {
            return null;
        }
        return this.items[this.front];
    }
    
    isEmpty() {
        return this.count === 0;
    }
    
    size() {
        return this.count;
    }
}

    int* items;     // Fixed-size array
    int front;      // Index of front element
    int rear;       // Index of rear element
    int count;      // Number of elements
    int capacity;

    Queue(int cap = 100) : capacity(cap), front(0), rear(-1), count(0) {
        items = new int[capacity];
    }
    
    ~Queue() { delete[] items; }
    
    // Add item to back - O(1)
    void enqueue(int item) {
        if (count >= capacity)
            throw std::overflow_error("Queue is full");
        rear = (rear + 1) % capacity;
        items[rear] = item;
        count++;
    }
    
    // Remove and return front item - O(1)
    int dequeue() {
        if (isEmpty())
            throw std::underflow_error("Queue is empty");
        int item = items[front];
        front = (front + 1) % capacity;
        count--;
        return item;
    }
    
    // View front item without removing - O(1)
    int peek() {
        if (isEmpty())
            throw std::underflow_error("Queue is empty");
        return items[front];
    }
    
    bool isEmpty() { return count == 0; }
    
    int size() { return count; }
};

#include <stdio.h>
#include <stdbool.h>

#define MAX_SIZE 100

typedef struct {
    int items[MAX_SIZE];  // Fixed-size array
    int front;            // Index of front element
    int rear;             // Index of rear element
    int count;            // Number of elements
} Queue;

void init(Queue* q) {
    q->front = 0;
    q->rear = -1;
    q->count = 0;
}

// Add item to back - O(1)
bool enqueue(Queue* q, int item) {
    if (q->count >= MAX_SIZE)
        return false;  // Queue overflow
    q->rear = (q->rear + 1) % MAX_SIZE;
    q->items[q->rear] = item;
    q->count++;
    return true;
}

// Remove and return front item - O(1)
int dequeue(Queue* q) {
    if (isEmpty(q))
        return -1;  // Queue underflow
    int item = q->items[q->front];
    q->front = (q->front + 1) % MAX_SIZE;
    q->count--;
    return item;
}

// View front item - O(1)
int peek(Queue* q) {
    if (isEmpty(q))
        return -1;
    return q->items[q->front];
}

bool isEmpty(Queue* q) {
    return q->count == 0;
}

int size(Queue* q) {
    return q->count;
}

// Usage
int main() {
    Queue queue;
    init(&queue);
    enqueue(&queue, 10);
    enqueue(&queue, 20);
    printf("%d", dequeue(&queue)); // 10
    return 0;
}
