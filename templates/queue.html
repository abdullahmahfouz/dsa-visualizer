Queue Operations

Size: 0
Front: Empty

Learn

Queue is a FIFO structure:

  • Enqueue O(1): Add to back
  • Dequeue O(n): Remove from front
  • Peek O(1): View front element

Example: Print queue, waiting lines

Stack vs Queue:

  • Stack = LIFO (plates)
  • Queue = FIFO (line at store)

class Queue:
    def __init__(self):
        self.items = []
    
    def enqueue(self, item):
        """Add item to back - O(1)"""
        self.items.append(item)
    
    def dequeue(self):
        """Remove and return front item - O(n)"""
        if not self.is_empty():
            return self.items.pop(0)
        raise IndexError("Queue is empty")
    
    def peek(self):
        """View front item without removing - O(1)"""
        if not self.is_empty():
            return self.items[0]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)

            return this.items.shift();
        }
        throw new Error("Queue is empty");
    }
    
    peek() {
        // View front item without removing - O(1)
        if (!this.isEmpty()) {
            return this.items[0];
        }
        return null;
    }
    
    isEmpty() {
        return this.items.length === 0;
    }
    
    size() {
        return this.items.length;
    }
}

import java.util.LinkedList;

public class Queue<T> {
    private LinkedList<T> items;
    
    public Queue() {
        this.items = new LinkedList<>();
    }
    
    // Add item to back - O(1)
    public void enqueue(T item) {
        items.addLast(item);
    }
    
    // Remove and return front item - O(1)
    public T dequeue() {
        if (!isEmpty()) {
            return items.removeFirst();
        }
        throw new RuntimeException("Queue is empty");
    }
    
    // View front item without removing - O(1)
    public T peek() {
        if (!isEmpty()) {
            return items.getFirst();
        }
        return null;
    }
    
    public boolean isEmpty() {
        return items.isEmpty();
    }
    
    public int size() {
        return items.size();
    }
}

public class Queue<T> {
    private LinkedList<T> items = new LinkedList<T>();
    
    // Add item to back - O(1)
    public void Enqueue(T item) {
        items.AddLast(item);
    }
    
    // Remove and return front item - O(1)
    public T Dequeue() {
        if (!IsEmpty()) {
            T item = items.First.Value;
            items.RemoveFirst();
            return item;
        }
        throw new InvalidOperationException("Queue is empty");
    }
    
    // View front item without removing - O(1)
    public T Peek() {
        if (!IsEmpty()) {
            return items.First.Value;
        }
        return default(T);
    }
    
    public bool IsEmpty() => items.Count == 0;
    
    public int Size() => items.Count;
}

#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

#define MAX_SIZE 100

typedef struct {
    int items[MAX_SIZE];
    int front;
    int rear;
    int count;
} Queue;

void init(Queue* q) {
    q->front = 0;
    q->rear = -1;
    q->count = 0;
}

// Add item to back - O(1)
void enqueue(Queue* q, int item) {
    if (q->count < MAX_SIZE) {
        q->rear = (q->rear + 1) % MAX_SIZE;
        q->items[q->rear] = item;
        q->count++;
    }
}

// Remove and return front item - O(1)
int dequeue(Queue* q) {
    if (!isEmpty(q)) {
        int item = q->items[q->front];
        q->front = (q->front + 1) % MAX_SIZE;
        q->count--;
        return item;
    }
    return -1; // Error
}

// View front item - O(1)
int peek(Queue* q) {
    if (!isEmpty(q)) {
        return q->items[q->front];
    }
    return -1;
}

bool isEmpty(Queue* q) {
    return q->count == 0;
}

// Usage
int main() {
    Queue queue;
    init(&queue);
    enqueue(&queue, 10);
    enqueue(&queue, 20);
    printf("%d", dequeue(&queue)); // 10
    return 0;
}
