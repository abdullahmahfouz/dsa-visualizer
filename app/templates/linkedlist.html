What is a Singly Linked List?

A Singly Linked List is a linear data structure where each element (called a node) contains data and a pointer to the next node in the sequence. Unlike arrays, linked lists do not store elements in contiguous memory locations, making insertions and deletions efficient at any position.

Think of a treasure hunt â€“ each clue (node) points to the next location!
Insert at Head Add a new node at the start O(1)
Insert at Tail Add a new node at the end O(n)
Insert at Index Add a new node at a specific position O(n)
Delete Head Remove the first node O(1)
Delete Tail Remove the last node O(n)
Delete at Index Remove a node at a specific position O(n)
Search Find a node by value O(n)
Common Uses:
  • Implementing stacks and queues
  • Navigation in web browsers (forward/backward history)
  • Dynamic memory allocation
  • Undo functionality in editors

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class SinglyLinkedList:
    def __init__(self):
        self.head = None
        self.size = 0
    
    def insert_at_head(self, data):
        """Insert at beginning - O(1)"""
        new_node = Node(data)
        new_node.next = self.head
        self.head = new_node
        self.size += 1
    
    def insert_at_tail(self, data):
        """Insert at end - O(n)"""
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            return
        current = self.head
        while current.next:
            current = current.next
        current.next = new_node
        self.size += 1
    
    def delete_at_head(self):
        """Delete first node - O(1)"""
        if not self.head:
            return None
        data = self.head.data
        self.head = self.head.next
        self.size -= 1
        return data
    
    def search(self, data):
        """Search for value - O(n)"""
        current = self.head
        index = 0
        while current:
            if current.data == data:
                return index
            current = current.next
            index += 1
        return -1

public class SinglyLinkedList {
    private Node head;
    private int size;
    
    public SinglyLinkedList() {
        head = null;
        size = 0;
    }
    
    // Insert at beginning - O(1)
    public void InsertAtHead(int data) {
        Node newNode = new Node(data);
        newNode.Next = head;
        head = newNode;
        size++;
    }
    
    // Insert at end - O(n)
    public void InsertAtTail(int data) {
        Node newNode = new Node(data);
        if (head == null) {
            head = newNode;
            return;
        }
        Node current = head;
        while (current.Next != null) {
            current = current.Next;
        }
        current.Next = newNode;
        size++;
    }
    
    // Delete first node - O(1)
    public int? DeleteAtHead() {
        if (head == null) return null;
        int data = head.Data;
        head = head.Next;
        size--;
        return data;
    }
    
    // Search for value - O(n)
    public int Search(int data) {
        Node current = head;
        int index = 0;
        while (current != null) {
            if (current.Data == data) return index;
            current = current.Next;
            index++;
        }
        return -1;
    }
    
    public int Size => size;
    public bool IsEmpty => head == null;
}

#include <iostream>

struct Node {
    int data;
    Node* next;
    Node(int d) : data(d), next(nullptr) {}
};

class SinglyLinkedList {
private:
    Node* head;
    int size;
    
public:
    SinglyLinkedList() : head(nullptr), size(0) {}
    
    // Insert at beginning - O(1)
    void insertAtHead(int data) {
        Node* newNode = new Node(data);
        newNode->next = head;
        head = newNode;
        size++;
    }
    
    // Insert at end - O(n)
    void insertAtTail(int data) {
        Node* newNode = new Node(data);
        if (!head) {
            head = newNode;
            return;
        }
        Node* current = head;
        while (current->next)
            current = current->next;
        current->next = newNode;
        size++;
    }
    
    // Delete first node - O(1)
    int deleteAtHead() {
        if (!head) return -1;
        int data = head->data;
        Node* temp = head;
        head = head->next;
        delete temp;
        size--;
        return data;
    }
    
    // Search for value - O(n)
    int search(int data) {
        Node* current = head;
        int index = 0;
        while (current) {
            if (current->data == data) return index;
            current = current->next;
            index++;
        }
        return -1;
    }
};

#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

typedef struct {
    Node* head;
    int size;
} SinglyLinkedList;

void init(SinglyLinkedList* list) {
    list->head = NULL;
    list->size = 0;
}

// Insert at beginning - O(1)
void insertAtHead(SinglyLinkedList* list, int data) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    newNode->data = data;
    newNode->next = list->head;
    list->head = newNode;
    list->size++;
}

// Insert at end - O(n)
void insertAtTail(SinglyLinkedList* list, int data) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    newNode->data = data;
    newNode->next = NULL;
    
    if (!list->head) {
        list->head = newNode;
        return;
    }
    Node* current = list->head;
    while (current->next)
        current = current->next;
    current->next = newNode;
    list->size++;
}

// Delete first node - O(1)
int deleteAtHead(SinglyLinkedList* list) {
    if (!list->head) return -1;
    int data = list->head->data;
    Node* temp = list->head;
    list->head = list->head->next;
    free(temp);
    list->size--;
    return data;
}

// Search for value - O(n)
int search(SinglyLinkedList* list, int data) {
    Node* current = list->head;
    int index = 0;
    while (current) {
        if (current->data == data) return index;
        current = current->next;
        index++;
    }
    return -1;
}

// Usage
int main() {
    SinglyLinkedList list;
    init(&list);
    insertAtHead(&list, 10);
    insertAtTail(&list, 20);
    printf("%d", search(&list, 20)); // 1
    return 0;
}
