What is a Hash Table with Linear Probing?

A Hash Table is a data structure that stores key-value pairs. It uses a hash function to compute an index into an array, allowing for fast lookups, insertions, and deletions. Linear Probing is a collision resolution technique where, if a slot is occupied, the algorithm checks the next slot in the array until an empty one is found.

Think of a library where books are sorted by genre (the hash) for faster retrieval.
Insert Add a key-value pair O(1) avg
Search Find a value by key O(1) avg
Delete Remove a key-value pair O(1) avg
Clear Remove all entries O(n)
Common Uses:
  • Database indexing
  • Implementing sets and maps
  • Caching (e.g., DNS cache)
  • Counting frequencies of items

Collisions and Linear Probing

A collision occurs when two different keys hash to the same index. Linear Probing is a simple way to resolve collisions. If a collision occurs, it checks the next available slot in the array and places the new key there. This process continues until an empty slot is found.

How the Hash Function Works

The hash function takes a key and converts it into an array index. A good hash function distributes keys evenly to minimize collisions.

Handling Collisions with Linear Probing

When two keys hash to the same index (a "collision"), linear probing finds the next empty slot.

"apple"
"banana"
→
hash(key) % 5 Both hash to index 2
→
"apple"
Index: 2 is taken
→
"apple"
"banana"
"banana" goes to next slot (3)
Note: Keys in a hash table must be unique. Inserting an existing key will update its value rather than create a duplicate.

class HashTable:
    DELETED = object()

    def __init__(self, size=11):
        self.size = size
        self.table = [None] * self.size
        self.count = 0

    @staticmethod
    def is_prime(n):
        if n <= 1:
            return False
        if n <= 3:
            return True
        if n % 2 == 0 or n % 3 == 0:
            return False
        i = 5
        while i * i <= n:
            if n % i == 0 or n % (i + 2) == 0:
                return False
            i += 6
        return True

    @staticmethod
    def next_prime(n):
        while not HashTable.is_prime(n):
            n += 1
        return n

    def _hash(self, key):
        return hash(key) % self.size

    def load_factor(self):
        return self.count / self.size

    def rehash(self):
        old_items = [item for item in self.table if item and item is not HashTable.DELETED]
        new_size = HashTable.next_prime(self.size * 2)
        self.size = new_size
        self.table = [None] * self.size
        self.count = 0
        for key, value in old_items:
            self.insert(key, value)

    def insert(self, key, value):
        if self.load_factor() > 0.7:
            self.rehash()
        idx = self._hash(key)
        start = idx
        while True:
            slot = self.table[idx]
            if slot is None or slot is HashTable.DELETED:
                self.table[idx] = (key, value)
                self.count += 1
                return
            if slot[0] == key:
                self.table[idx] = (key, value)  # update
                return
            idx = (idx + 1) % self.size
            if idx == start:
                raise RuntimeError('HashTable is full')

    def search(self, key):
        idx = self._hash(key)
        start = idx
        while True:
            slot = self.table[idx]
            if slot is None:
                return None
            if slot is not HashTable.DELETED and slot[0] == key:
                return slot[1]
            idx = (idx + 1) % self.size
            if idx == start:
                return None

    def delete(self, key):
        idx = self._hash(key)
        start = idx
        while True:
            slot = self.table[idx]
            if slot is None:
                return
            if slot is not HashTable.DELETED and slot[0] == key:
                self.table[idx] = HashTable.DELETED
                self.count -= 1
                return
            idx = (idx + 1) % self.size
            if idx == start:
                return

class HashTable {
    constructor(size = 11) {
        this.size = size;
        this.table = new Array(size).fill(null);
        this.count = 0;
    }

    static isPrime(n) {
        if (n <= 1) return false;
        if (n <= 3) return true;
        if (n % 2 === 0 || n % 3 === 0) return false;
        for (let i = 5; i * i <= n; i += 6) {
            if (n % i === 0 || n % (i + 2) === 0) return false;
        }
        return true;
    }

    static nextPrime(n) {
        while (!HashTable.isPrime(n)) n++;
        return n;
    }

    _hash(key) {
        let h = 0;
        const s = String(key);
        for (let i = 0; i < s.length; i++) h = (h * 31 + s.charCodeAt(i)) % this.size;
        return h;
    }

    rehash() {
        const items = this.table.filter(x => x !== null && x !== 'DELETED');
        const newSize = HashTable.nextPrime(this.size * 2);
        this.size = newSize;
        this.table = new Array(this.size).fill(null);
        this.count = 0;
        for (const [k, v] of items) this.insert(k, v);
    }

    insert(key, value) {
        if (this.count / this.size > 0.7) this.rehash();
        let idx = this._hash(key);
        const start = idx;
        while (true) {
            const slot = this.table[idx];
            if (slot === null || slot === 'DELETED') {
                this.table[idx] = [key, value];
                this.count++;
                return;
            }
            if (slot[0] === key) {
                this.table[idx][1] = value; // update
                return;
            }
            idx = (idx + 1) % this.size;
            if (idx === start) throw new Error('HashTable full');
        }
    }

    search(key) {
        let idx = this._hash(key);
        const start = idx;
        while (true) {
            const slot = this.table[idx];
            if (slot === null) return null;
            if (slot !== 'DELETED' && slot[0] === key) return slot[1];
            idx = (idx + 1) % this.size;
            if (idx === start) return null;
        }
    }

    delete(key) {
        let idx = this._hash(key);
        const start = idx;
        while (true) {
            const slot = this.table[idx];
            if (slot === null) return;
            if (slot !== 'DELETED' && slot[0] === key) {
                this.table[idx] = 'DELETED';
                this.count--;
                return;
            }
            idx = (idx + 1) % this.size;
            if (idx === start) return;
        }
    }
}

import java.util.Arrays;

class HashTable {
    private static final int DEFAULT_CAPACITY = 10;
    private Entry[] table;
    private int size;

    static class Entry {
        Object key;
        Object value;
        boolean isDeleted;

        Entry(Object key, Object value) {
            this.key = key;
            this.value = value;
            this.isDeleted = false;
        }
    }

    public HashTable() {
        table = new Entry[DEFAULT_CAPACITY];
        size = 0;
    }

    private int hash(Object key) {
        return Math.abs(key.hashCode() % table.length);
    }

    public void insert(Object key, Object value) {
        int index = hash(key);
        int startIndex = index;
        while (table[index] != null && !table[index].isDeleted) {
            if (table[index].key.equals(key)) {
                table[index].value = value; // Update existing key
                return;
            }
            index = (index + 1) % table.length;
            if (index == startIndex) { // Table is full
                System.err.println("Hash table is full.");
                return;
            }
        }
        table[index] = new Entry(key, value);
        size++;
    }

    public Object search(Object key) {
        int index = hash(key);
        int startIndex = index;
        while (table[index] != null) {
            if (!table[index].isDeleted && table[index].key.equals(key)) {
                return table[index].value;
            }
            index = (index + 1) % table.length;
            if (index == startIndex) { // Cycled through table
                break;
            }
        }
        return null;
    }

    public void delete(Object key) {
        int index = hash(key);
        int startIndex = index;
        while (table[index] != null) {
            if (!table[index].isDeleted && table[index].key.equals(key)) {
                table[index].isDeleted = true;
                size--;
                return;
            }
            index = (index + 1) % table.length;
            if (index == startIndex) { // Cycled through table
                break;
            }
        }
    }
}

using System;
using System.Linq;

public class HashTable
{
    private class Entry
    {
        public object Key;
        public object Value;
        public bool IsDeleted = false;
        public Entry(object k, object v) { Key = k; Value = v; }
    }

    private Entry[] _table;
    private int _count = 0;

    public HashTable(int capacity = 11) { _table = new Entry[capacity]; }

    private int Hash(object key) => Math.Abs(key.GetHashCode() % _table.Length);

    private static bool IsPrime(int n) {
        if (n <= 1) return false; if (n <= 3) return true; if (n % 2 == 0 || n % 3 == 0) return false;
        for (int i = 5; i * i <= n; i += 6) if (n % i == 0 || n % (i + 2) == 0) return false;
        return true;
    }

    private static int NextPrime(int n) { while (!IsPrime(n)) n++; return n; }

    private void Rehash() {
        var old = _table.Where(e => e != null && !e.IsDeleted).ToArray();
        _table = new Entry[NextPrime(_table.Length * 2)]; _count = 0;
        foreach (var e in old) Insert(e.Key, e.Value);
    }

    public void Insert(object key, object value) {
        if ((double)_count / _table.Length > 0.7) Rehash();
        int idx = Hash(key), start = idx;
        while (true) {
            var e = _table[idx];
            if (e == null || e.IsDeleted) { _table[idx] = new Entry(key, value); _count++; return; }
            if (e.Key.Equals(key)) { e.Value = value; return; }
            idx = (idx + 1) % _table.Length; if (idx == start) throw new InvalidOperationException("Full");
        }
    }

    public object Search(object key) {
        int idx = Hash(key), start = idx;
        while (_table[idx] != null) {
            var e = _table[idx]; if (!e.IsDeleted && e.Key.Equals(key)) return e.Value;
            idx = (idx + 1) % _table.Length; if (idx == start) break;
        }
        return null;
    }

    public void Delete(object key) {
        int idx = Hash(key), start = idx;
        while (_table[idx] != null) {
            var e = _table[idx]; if (!e.IsDeleted && e.Key.Equals(key)) { e.IsDeleted = true; _count--; return; }
            idx = (idx + 1) % _table.Length; if (idx == start) break;
        }
    }
}

template<typename K, typename V>
class HashTable {
private:
    struct Entry {
        K key;
        V value;
        bool is_deleted = false;
    };
    std::vector<std::optional<Entry>> table;
    size_t table_size;
    size_t count = 0;

    size_t hash(const K& key) {
        return std::hash<K>{}(key) % table_size;
    }

public:
    HashTable(size_t size = 10) : table_size(size) {
        table.resize(table_size);
    }

    void insert(const K& key, const V& value) {
        if (count >= table_size) {
            // In a real implementation, you would resize the table here
            std::cerr << "Table is full!" << std::endl;
            return;
        }
        size_t index = hash(key);
        while (table[index] && !table[index]->is_deleted) {
            if (table[index]->key == key) {
                table[index]->value = value; // Update
                return;
            }
            index = (index + 1) % table_size;
        }
        table[index] = Entry{key, value};
        count++;
    }

    std::optional<V> search(const K& key) {
        size_t index = hash(key);
        size_t start_index = index;
        while (table[index]) {
            if (!table[index]->is_deleted && table[index]->key == key) {
                return table[index]->value;
            }
            index = (index + 1) % table_size;
            if (index == start_index) break;
        }
        return std::nullopt;
    }

    void remove(const K& key) {
        size_t index = hash(key);
        size_t start_index = index;
        while (table[index]) {
            if (!table[index]->is_deleted && table[index]->key == key) {
                table[index]->is_deleted = true;
                count--;
                return;
            }
            index = (index + 1) % table_size;
            if (index == start_index) break;
        }
    }
};

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define TABLE_SIZE 10

typedef struct {
    char* key;
    int value;
    int is_deleted;
} Entry;

Entry* table[TABLE_SIZE];

unsigned int hash(char* key) {
    unsigned int hash = 0;
    while (*key) {
        hash = (hash << 5) + *key++;
    }
    return hash % TABLE_SIZE;
}

void init_table() {
    for (int i = 0; i < TABLE_SIZE; i++) {
        table[i] = NULL;
    }
}

void insert(char* key, int value) {
    unsigned int index = hash(key);
    int original_index = index;
    while (table[index] != NULL && !table[index]->is_deleted) {
        if (strcmp(table[index]->key, key) == 0) {
            table[index]->value = value; // Update
            return;
        }
        index = (index + 1) % TABLE_SIZE;
        if (index == original_index) {
            printf("Error: Hash table is full.\n");
            return;
        }
    }
    table[index] = (Entry*)malloc(sizeof(Entry));
    table[index]->key = strdup(key);
    table[index]->value = value;
    table[index]->is_deleted = 0;
}

Entry* search(char* key) {
    unsigned int index = hash(key);
    int original_index = index;
    while (table[index] != NULL) {
        if (!table[index]->is_deleted && strcmp(table[index]->key, key) == 0) {
            return table[index];
        }
        index = (index + 1) % TABLE_SIZE;
        if (index == original_index) break;
    }
    return NULL;
}

void delete_entry(char* key) {
    unsigned int index = hash(key);
    int original_index = index;
    while (table[index] != NULL) {
        if (!table[index]->is_deleted && strcmp(table[index]->key, key) == 0) {
            table[index]->is_deleted = 1;
            return;
        }
        index = (index + 1) % TABLE_SIZE;
        if (index == original_index) break;
    }
}
